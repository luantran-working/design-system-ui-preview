API

Simple, Powerful API

Production-ready in minutes with our intuitive SDK

import cohere

co = cohere.Client('your-api-key')

# Generate with Command
response = co.generate(
  model='command',
  prompt='Write a product description for enterprise AI',
  max_tokens=300,
  temperature=0.7
)

print(response.generations[0].text)

# Semantic search with Embed
docs = ["AI transforms business", "Machine learning at scale"]
query = "enterprise AI solutions"

embeddings = co.embed(
  texts=docs + [query],
  model='embed-english-v3.0'
)

# Rerank for precision
results = co.rerank(
  query=query,
  documents=docs,
  model='rerank-english-v2.0',
  top_n=3
)
